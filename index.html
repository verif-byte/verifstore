// ===================================================== // VERIF STORE - ULTIMATE FULL PRODUCTION (ALL-IN-ONE) // Stack: Node.js + Express + MongoDB + JWT + Admin Panel // Single File Version (Simple Deployment) // ===================================================== // ================= INSTALL ================= // 1. Install Node.js // 2. npm init -y // 3. npm install express mongoose bcryptjs jsonwebtoken cors dotenv uuid axios // 4. Create .env file: // MONGO_URI=your_mongodb_connection // JWT_SECRET=supersecretkey // ADMIN_SECRET=adminaccesskey // DISCORD_WEBHOOK=optional_webhook_url // 5. node server.js // =========================================== require('dotenv').config() const express = require('express') const mongoose = require('mongoose') const bcrypt = require('bcryptjs') const jwt = require('jsonwebtoken') const cors = require('cors') const { v4: uuidv4 } = require('uuid') const axios = require('axios') const app = express() app.use(cors()) app.use(express.json()) // ================= DATABASE ================= mongoose.connect(process.env.MONGO_URI) .then(()=>console.log("MongoDB Connected")) .catch(err=>console.log(err)) // ================= MODELS ================= const User = mongoose.model('User', new mongoose.Schema({ username:String, email:String, password:String, role:{ type:String, default:"user" } })) const Product = mongoose.model('Product', new mongoose.Schema({ name:String, game:String, price:Number, stock:{ type:Number, default:9999 } })) const Order = mongoose.model('Order', new mongoose.Schema({ orderId:String, userId:String, items:Array, total:Number, paymentMethod:String, paymentStatus:{ type:String, default:"UNPAID" }, status:{ type:String, default:"PENDING" }, createdAt:{ type:Date, default:Date.now } })) // ================= MIDDLEWARE ================= function auth(req,res,next){ const token = req.headers.authorization if(!token) return res.status(401).json({msg:"No Token"}) try{ const decoded = jwt.verify(token,process.env.JWT_SECRET) req.user = decoded next() }catch{ res.status(401).json({msg:"Invalid Token"}) } } function adminAuth(req,res,next){ if(req.headers.adminsecret !== process.env.ADMIN_SECRET) return res.status(403).json({msg:"Admin access denied"}) next() } // ================= AUTH ================= app.post('/api/register', async(req,res)=>{ const {username,email,password} = req.body const hashed = await bcrypt.hash(password,10) await User.create({username,email,password:hashed}) res.json({msg:"Registered"}) }) app.post('/api/login', async(req,res)=>{ const {email,password} = req.body const user = await User.findOne({email}) if(!user) return res.status(400).json({msg:"User not found"}) const valid = await bcrypt.compare(password,user.password) if(!valid) return res.status(400).json({msg:"Wrong password"}) const token = jwt.sign({id:user._id,role:user.role},process.env.JWT_SECRET) res.json({token}) }) // ================= PRODUCTS ================= app.get('/api/products', async(req,res)=>{ res.json(await Product.find()) }) app.post('/api/admin/add-product', adminAuth, async(req,res)=>{ await Product.create(req.body) res.json({msg:"Product Added"}) }) // ================= CHECKOUT ================= app.post('/api/checkout', auth, async(req,res)=>{ const {items,paymentMethod} = req.body const total = items.reduce((sum,i)=>sum+i.price,0) const order = await Order.create({ orderId:uuidv4(), userId:req.user.id, items, total, paymentMethod }) // Discord Notification (Optional) if(process.env.DISCORD_WEBHOOK){ await axios.post(process.env.DISCORD_WEBHOOK,{ content:`New Order ${order.orderId} - Rp ${total}` }) } res.json({ msg:"Order Created", orderId:order.orderId, paymentInstruction:`Pay ${total} via ${paymentMethod}` }) }) // ================= PAYMENT CONFIRM (SIMULATED) ================= app.post('/api/payment-confirm', async(req,res)=>{ const {orderId} = req.body await Order.findOneAndUpdate({orderId},{paymentStatus:"PAID",status:"PROCESSING"}) res.json({msg:"Payment Confirmed"}) }) // ================= ADMIN DASHBOARD DATA ================= app.get('/api/admin/orders', adminAuth, async(req,res)=>{ res.json(await Order.find().sort({createdAt:-1})) }) // ================= FRONTEND ================= app.get('/',(req,res)=>{ res.send(` Verif Store Ultimate

    `) }) app.listen(3000,()=>console.log("Verif Store Ultimate running on http://localhost:3000"))
